4. Exercice#

  • Créer des catégories d’ages avec la fonction map() de pandas

  • Créer des catégories de genres avec cat.codes

def category_ages(age):
    if age <= 20:
        return '<20 ans'
    elif (age > 20) & (age <= 30):
        return '20-30 ans'
    elif (age > 30) & (age <= 40):
        return '30-40 ans'
    else:
        return '+40 ans'

data['age'] = data['age'].map(category_ages)
